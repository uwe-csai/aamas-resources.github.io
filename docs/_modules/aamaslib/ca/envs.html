"""
aamaslib.ca.envs
================

Generic Cellular Automaton environment that is rule-agnostic and accepts pluggable update rules.
Designed to be simple to use in AAMAS worksheets.

Overview
--------
- Rule-agnostic: you can plug any cell-update function.
- Neighbourhoods and boundary handling use the utilities provided by aamaslib.ca.
- Useful for small experiments and educational purposes.

Example
-------
Basic usage with a simple "Game of Life"-like rule:

    import numpy as np
    from aamaslib.ca import moore, boundary_dead, GridRenderer
    from aamaslib.ca.envs import CellularEnvironment

    def life_rule(grid, i, j, neighbours):
        # classic Conway
        if grid[i, j] == 1:
            return 1 if neighbours in (2, 3) else 0
        else:
            return 1 if neighbours == 3 else 0

    config = {
        "grid": [
            "....................",
            "....................",
            "....OOO.............",
            "....O...............",
            ".....O..............",
            "....................",
            "....................",
            "....................",
            "....................",
            "...................."
        ],
        "max_iterations": 50,
        "neighbourhood": {"type": "moore", "radius": 1},
        "boundary": {"type": "dead"},
        "rule": {"type": "conway"}  # used only if you also pass a rule_registry
    }

    renderer = GridRenderer(mode="display", pause_ms=100)
    env = CellularEnvironment(config, render_mode="human", renderer=renderer, update_rule=life_rule)

    grid = env.reset()
    terminated = truncated = False
    while not (terminated or truncated):
        grid, terminated, truncated, info = env.step()
        env.render()

    env.close()
"""

class CellularEnvironment:
    """
    Generic cellular automaton environment with pluggable rules.

    This environment evolves a binary grid (0 = dead, 1 = alive) using a provided
    update rule and configurable neighborhood and boundary conditions.

    Parameters
    ----------
    config : dict
        Configuration dictionary. Expected keys:
        - grid: list[str]
            Initial grid as a list of strings where 'O' is alive and any other
            character is treated as dead.
        - max_iterations: int
            Maximum number of steps before truncation.
        - neighbourhood: dict
            {
              "type": "moore" | "von_neumann" | "custom",
              "radius": int (for moore or von_neumann),
              "custom_offsets": list[tuple[int, int]] (if type == "custom")
            }
        - boundary: dict
            {"type": "dead" | "wrap" | "mirror"}
        - rule: dict
            {"type": "<name>"} used only if you also pass a rule_registry.
    env_shape : str, default="grid"
        Currently supports only "grid" (directly uses config["grid"]).
    rule_registry : dict[str, Callable], optional
        Optional registry for looking up a rule by name in config["rule"]["type"].
        If provided and no explicit update_rule is passed, the environment will
        resolve the rule via this registry.
    render_mode : str, optional
        If "human" and a renderer is provided, render() will display the grid.
    renderer : Any, optional
        An object compatible with aamaslib.ca.rendering.GridRenderer:
        must implement .show(grid, iteration, update_rule_name) and .close().
    update_rule : Callable, optional
        Explicitly provide the rule callable. If provided, it overrides rule_registry.

    Notes
    -----
    - The environment is intentionally minimal and not tied to Gymnasium APIs,
      but it follows a similar pattern: reset() and step() returning
      (observation, terminated, truncated, info).
    - The observation is the current grid (np.ndarray[int]).
    """

    @classmethod
    def from_yaml(
        cls,
        yaml_path: str,
        *,
        overrides: Optional[Dict[str, Any]] = None,
        **kwargs: Any,
    ) -> "CellularEnvironment":
        """
        Construct an environment from a YAML configuration file.

        Parameters
        ----------
        yaml_path : str
            Path to a YAML file containing the configuration (see class docstring).
        overrides : dict, optional
            Dict of keys to update/override the loaded configuration.
        **kwargs
            Additional arguments forwarded to the class constructor (e.g. renderer, update_rule).

        Returns
        -------
        CellularEnvironment
            Configured environment instance.
        """
        with open(yaml_path, "r") as f:
            config = yaml.safe_load(f) or {}
        if overrides:
            # shallow update is sufficient for typical keys; deep-merge if needed
            config.update(overrides)
        return cls(config, **kwargs)



    def __init__(
        self,
        config: Dict[str, Any],
        env_shape: str = "grid",
        rule_registry: Optional[Dict[str, Callable[[np.ndarray, int, int, int], int]]] = None,
        render_mode: Optional[str] = None,
        renderer: Optional[Any] = None,
        update_rule: Optional[Union[UpdateRuleProtocol, Callable[[np.ndarray, int, int, int], int]]] = None,
    ) -> None:
        # --- Resolve rule ---
        if update_rule is not None:
            self.update_rule = update_rule
        else:
            rule_cfg = (config or {}).get("rule", {})
            rule_name = rule_cfg.get("type")
            if rule_name and rule_registry:
                try:
                    self.update_rule = rule_registry[rule_name]
                except KeyError:
                    raise ValueError(f"Rule '{rule_name}' not found in provided rule_registry.")
            else:
                raise ValueError(
                    "No update_rule provided and no resolvable rule via rule_registry. "
                    "Provide `update_rule=` or pass a `rule_registry` and set config['rule']['type']."
                )

        # --- Parse initial grid ---
        if env_shape != "grid":
            raise ValueError(f"Unsupported env_shape: {env_shape!r}. Only 'grid' is supported.")
        try:
            symbol_grid = config["grid"]
        except KeyError:
            raise KeyError("config['grid'] is required (list[str] with 'O' for alive).")
        if not isinstance(symbol_grid, list) or not all(isinstance(r, str) for r in symbol_grid):
            raise TypeError("config['grid'] must be a list of strings.")

        self.initial_grid = np.array(
            [[1 if cell == "O" else 0 for cell in row] for row in symbol_grid],
            dtype=np.int8,
        )
        self.height, self.width = self.initial_grid.shape

        # --- Iteration limits ---
        try:
            self.max_iterations = int(config["max_iterations"])
        except KeyError:
            raise KeyError("config['max_iterations'] is required (int).")

        # --- Neighborhood ---
        nb_cfg = config.get("neighbourhood", {}) or {}
        nb_type = nb_cfg.get("type", "moore")
        nb_rad = int(nb_cfg.get("radius", 1))
        nb_cust = nb_cfg.get("custom_offsets", None)

        if nb_type not in NEIGHBOURHOODS and nb_type != "custom":
            raise ValueError(f"Unknown neighbourhood: {nb_type!r}")
        if nb_type == "custom":
            if nb_cust is None:
                raise ValueError("Custom neighbourhood requires 'custom_offsets'.")
            self._neighbour_offsets = NEIGHBOURHOODS["custom"](nb_cust)
        else:
            self._neighbour_offsets = NEIGHBOURHOODS[nb_type](nb_rad)

        # --- Boundary ---
        bd_cfg = config.get("boundary", {}) or {}
        bd_type = bd_cfg.get("type", "dead")
        if bd_type not in BOUNDARIES:
            raise ValueError(f"Unknown boundary: {bd_type!r}")
        self._boundary_fn = BOUNDARIES[bd_type](self.height, self.width)

        # --- State ---
        self.grid: Optional[np.ndarray] = None
        self.iteration: int = 0

        # --- Rendering ---
        self.render_mode = render_mode  # "human" or None
        self.renderer = renderer        # e.g., an instance of aamaslib.ca.rendering.GridRenderer

    def reset(self) -> np.ndarray:
        """
        Reset environment to the initial grid.

        Returns

        np.ndarray

        """
        self.grid = self.initial_grid.copy()
        self.iteration = 0

    def step(self) -> Tuple[np.ndarray, bool, bool, Dict[str, Any]]:
        """
        Advance the environment by one step using the update rule.

        Returns
        -------
        observation : np.ndarray
            The new grid after applying the rule.
        terminated : bool
            True if a terminal condition is reached (e.g., all cells dead).
        truncated : bool
            True if max_iterations has been reached.
        info : dict
            Additional information about the state (iteration, live_cells, rule name).
        """
        if self.grid is None:
            raise RuntimeError("Call reset() before step().")

        truncated = self.iteration >= self.max_iterations
        if truncated:
            return self.grid.copy(), False, True, {"reason": "max_iterations", **self.get_state()}

        # Apply update rule to each cell
        new_grid = np.zeros_like(self.grid)
        for i in range(self.height):
            for j in range(self.width):
                neighbours = self._count_neighbours(i, j)
                new_grid[i, j] = self.update_rule(self.grid, i, j, neighbours)

        self.grid = new_grid
        self.iteration += 1

        terminated = self._is_terminated()
        info = self.get_state()
        return self.grid.copy(), terminated, truncated, info



    # --------------------------------------------------------------------- #
    # Helpers
    # --------------------------------------------------------------------- #

    def _count_neighbours(self, row: int, col: int) -> int:
        """
        Count live neighbours for a given cell using the configured neighborhood and boundary.

        Parameters
        ----------
        row : int
            Row index of the cell.
        col : int
            Column index of the cell.

        Returns
        -------
        int
            Number of live neighbours.
        """
        assert self.grid is not None, "Grid must be initialized (call reset())."
        count = 0
        for di, dj in self._neighbour_offsets:
            ni, nj, valid = self._boundary_fn(row, col, di, dj)
            if not valid:
                continue
            count += self.grid[ni, nj]
        return int(count)

    def _is_terminated(self) -> bool:
        """
        Check for terminal condition.

        Returns
        -------
        bool
            True if all cells are dead; otherwise False.
        """
        assert self.grid is not None, "Grid must be initialized (call reset())."
        return int(np.sum(self.grid)) == 0

        return (self.height, self.width)

    def get_state(self) -> Dict[str, Any]:
        """
        Get current environment state information.

        Returns
        -------
        dict
            {
              'grid': np.ndarray,
              'iteration': int,
              'live_cells': int,
              'rule_name': str
            }
        """
        assert self.grid is not None, "Grid must be initialized (call reset())."
        rule_name = getattr(self.update_rule, "name", None)
        if rule_name is None:
            # Fall back to function name or repr
            rule_name = getattr(self.update_rule, "__name__", repr(self.update_rule))
        return {
            "grid": self.grid.copy(),
            "iteration": self.iteration,
            "live_cells": int(np.sum(self.grid)),
            "rule_name": rule_name,
        }

    def render(self) -> None:
        """
        Render the current grid state through the configured renderer (if any).
        """
        if self.render_mode == "human" and self.renderer is not None and self.grid is not None:
            rule_name = getattr(self.update_rule, "name", getattr(self.update_rule, "__name__", "rule"))
            self.renderer.show(self.grid, self.iteration, rule_name)

    def close(self) -> None:
        """
        Clean up renderer resources (if any).
        """
        if self.renderer is not None:
            close_fn = getattr(self.renderer, "close", None)
            if callable(close_fn):
                close_fn()

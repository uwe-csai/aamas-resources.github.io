"""
Grid rendering utilities for cellular automata visualization.

This module provides the GridRenderer class for displaying cellular automata
grids either in real-time or as animated GIFs.
"""

import matplotlib.pyplot as plt
from matplotlib.backends.backend_agg import FigureCanvasAgg
import numpy as np
from typing import Optional, List

try:
    import imageio.v2 as imageio
except ImportError:
    import imageio

class GridRenderer:
    """
    A renderer for cellular automata grids.
  
    Supports both real-time display and GIF generation modes.
    """
  
    def __init__(self,
                 mode: str = "display",    # "display" or "gif"
                 cmap: str = "gray_r",
                 vmin: int = 0,
                 vmax: int = 1,
                 title_prefix: str = "Iteration",
                 pause_ms: int = 20,    # only used in display mode
                 fps: int = 10    # only used in gif mode
                 ):
        """
        Initialize the GridRenderer.
      
        Args:
            mode: "display" for real-time or "gif" for animation generation
            cmap: Matplotlib colormap name
            vmin: Minimum value for color mapping
            vmax: Maximum value for color mapping
            title_prefix: Prefix for plot titles
            pause_ms: Pause duration in milliseconds for display mode
            fps: Frames per second for GIF mode
        """
        self.mode = mode
        self.cmap = cmap
        self.vmin = vmin
        self.vmax = vmax
        self.title_prefix = title_prefix

        # Timing knobs
        self.pause_ms = max(0, int(pause_ms))
        self.fps = max(1, int(fps))

        # Display state
        self._fig: Optional[plt.Figure] = None
        self._ax: Optional[plt.Axes] = None
        self._im = None
        self._interactive_enabled = False

        # GIF state
        self._frames: List[np.ndarray] = []

    def _ensure_interactive(self):
        """Enable interactive plotting if not already enabled."""
        if not self._interactive_enabled:
            plt.ion()
            self._interactive_enabled = True

    def _show_live(self, grid: np.ndarray, iteration: int, update_rule: str):
        """Display grid in real-time mode."""
        self._ensure_interactive()
        if self._fig is None:
            self._fig, self._ax = plt.subplots()
            self._im = self._ax.imshow(grid, cmap=self.cmap, vmin=self.vmin, vmax=self.vmax, interpolation="nearest")
            self._ax.set_xticks([])
            self._ax.set_yticks([])
            if hasattr(self._fig.canvas, "toolbar_visible"):
                self._fig.canvas.toolbar_visible = False
        else:
            self._im.set_data(grid)
          
        self._ax.set_title(f"{update_rule} - {self.title_prefix} {iteration}")

        # This pause controls perceived FPS in display mode
        self._fig.canvas.draw_idle()
        plt.pause(self.pause_ms / 1000.0)

    def _capture_frame(self, grid: np.ndarray, iteration: int, update_rule: str):
        """Capture frame for GIF generation."""
        # Render to off-screen Agg buffer for backend-agnostic frame capture
        fig, ax = plt.subplots(figsize=(4, 4))
        try:
            ax.imshow(grid, cmap=self.cmap, vmin=self.vmin, vmax=self.vmax, interpolation="nearest")
            ax.set_title(f"{update_rule} - {self.title_prefix} {iteration}")
            ax.set_xticks([])
            ax.set_yticks([])

            canvas = FigureCanvasAgg(fig)
            canvas.draw()
            w, h = fig.canvas.get_width_height()
            buf = np.frombuffer(canvas.buffer_rgba(), dtype=np.uint8).reshape(h, w, 4)
            rgb = buf[..., :3]  # drop alpha
            self._frames.append(rgb.copy())
        finally:
            plt.close(fig)

    def show(self, grid: np.ndarray, iteration: int = 0, update_rule: str = ""):
        """
        Display or capture a grid frame.
      
        Args:
            grid: 2D numpy array representing the grid
            iteration: Current iteration number
            update_rule: Name of the update rule being used
        """
        if grid is None:
            return
        if self.mode == "display":
            self._show_live(grid, iteration, update_rule)
        elif self.mode == "gif":
            self._capture_frame(grid, iteration, update_rule)
        else:
            raise ValueError("mode must be 'display' or 'gif'")

    def save_gif(self, path: str):
        """
        Save captured frames as a GIF.
      
        Args:
            path: Output file path for the GIF
        """
        if self.mode != "gif":
            raise ValueError("save_gif() only works in 'gif' mode")
        if not self._frames:
            raise ValueError("No frames captured. Call show() before save_gif().")

        # fps controls GIF timing precisely here
        duration = 1.0 / float(self.fps)
        imageio.mimsave(path, self._frames, duration=duration)

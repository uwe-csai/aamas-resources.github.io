"""
Boundary condition functions for cellular automata.

This module provides different boundary condition strategies for handling
cells at the edges of the grid.
"""

from typing import Callable, Tuple

def boundary_dead(height: int, width: int) -> Callable[[int, int, int, int], Tuple[int, int, bool]]:
    """
    Create a dead boundary condition function.
  
    Cells outside the grid are considered dead/inactive.
  
    Args:
        height: Grid height
        width: Grid width
      
    Returns:
        Boundary function that returns (row, col, is_valid)
    """
    def fn(row: int, col: int, di: int, dj: int) -> Tuple[int, int, bool]:
        ni, nj = row + di, col + dj
        if 0 <= ni < height and 0 <= nj < width:
            return ni, nj, True
        return 0, 0, False
    return fn

def boundary_wrap(height: int, width: int) -> Callable[[int, int, int, int], Tuple[int, int, bool]]:
    """
    Create a wrap-around boundary condition function.
  
    The grid wraps around at the edges (toroidal topology).
  
    Args:
        height: Grid height
        width: Grid width
      
    Returns:
        Boundary function that returns (row, col, is_valid)
    """
    def fn(row: int, col: int, di: int, dj: int) -> Tuple[int, int, bool]:
        ni, nj = row + di, col + dj
        return ni % height, nj % width, True
    return fn

def boundary_mirror(height: int, width: int) -> Callable[[int, int, int, int], Tuple[int, int, bool]]:
    """
    Create a mirror boundary condition function.
  
    Coordinates outside the grid are reflected back into the grid.
  
    Args:
        height: Grid height
        width: Grid width
      
    Returns:
        Boundary function that returns (row, col, is_valid)
    """
    def fn(row: int, col: int, di: int, dj: int) -> Tuple[int, int, bool]:
        ni, nj = row + di, col + dj
        ni = max(0, min(height - 1, ni))
        nj = max(0, min(width - 1, nj))
        return ni, nj, True
    return fn

BOUNDARIES = {
    "dead": boundary_dead,
    "wrap": boundary_wrap,
    "mirror": boundary_mirror
